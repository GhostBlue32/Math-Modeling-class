function ploty(x0,h,ends,linecolor,xname)
tv=linspace(0,ends,ends/h+1);
xv=linspace(0,ends,ends/h+1);
tv(1)=0;
xv(1)=x0;
for i=2:ends/h+1
    tv(i)=h*(i-1);
    xv(i)=xv(i-1)+h*cal(xv(i-1));
end
plot(tv,xv,'Color',linecolor,'DisplayName',xname)
end

function xpr=cal(x)
xpr=x*(1-x);
end
